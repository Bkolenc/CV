Une petite boîte à outils simplifiée qui s'ouvre et fait apparaître différents mots clés, développement, journalisme, qui eux même se développent

CREATE TABLE experiences (
    idExp int NOT NULL AUTO_INCREMENT,
    idLieu int NOT NULL,
	intitule varchar,
    debut date,
	fin date,
    PRIMARY KEY (idExp),
    FOREIGN KEY (idLieu) REFERENCES lieu(idLieu)
);

CREATE TABLE missions (
	idmission int NOT NULL AUTO_INCREMENT,
	idexp int NOT NULL,
	textemission text,
	PRIMARY KEY (idmission),
	FOREIGN KEY (idexp) REFERENCES experience(idexp)
);
